multnomah_pep_2016 <- get_estimates(
  geography = "county",
  state = "OR",
  county = "Multnomah",
  product = "characteristics",
  breakdown = c("SEX", "AGEGROUP"),
  breakdown_labels = TRUE,
  year = 2016
)

multnomah_pep_filtered <- multnomah_pep_2016 %>%
  filter(str_detect(AGEGROUP, "^Age"),
         SEX != "Both sexes") %>%
  mutate(value = ifelse(SEX == "Male", -value, value))

multnomah_pyramid <- ggplot(multnomah_pep_filtered,
                           aes(x = value,
                               y = AGEGROUP,
                               fill = SEX)) +
  geom_col(width = 0.95, alpha = 0.75) +
  theme_minimal(base_family = "Verdana", base_size = 12) +
  scale_x_continuous(
    labels = ~ number_format(scale = .001, suffix = "k")(abs(.x)),
    limits = 45000 * c(-1,1)
  ) +
  scale_y_discrete(labels = ~ str_remove_all(.x, "Age\\s|\\syears")) +
  scale_fill_manual(values = c("deeppink", "deepskyblue")) +
  labs(x = "",
       y = "2016 Census Bureau\n Population Estimate",
       title = "Population structure in Multnomah County, OR",
       fill = "",
       caption = "Sources: US Census Bureau PEP, tidycensus R package") +
  theme_minimal(base_family = "Verdana", base_size = 12) +
  theme(
    plot.title = element_text(hjust = 0.5, face = "bold", margin = margin(b = 15)),
    axis.title.y = element_text(margin = margin(r = 25, l = 10), vjust = 0.5),
    plot.margin = margin(t = 20, r = 25, b = 20, l = 25, unit = "pt")
  )

multnomah_local_seg <- multnomah_acs_data %>%
  mutual_local(
    group = "variable",
    unit = "GEOID",
    weight = "estimate",
    wide = TRUE
  )


multnomah_tracts <- tracts("OR", county = "Multnomah", cb = TRUE, year = 2016)

multnomah_tracts_seg <- multnomah_tracts %>%
  inner_join(multnomah_local_seg, by = "GEOID")

multnomah_tracts_seg_plot <- multnomah_tracts_seg %>%
  ggplot(aes(fill = ls)) +
  geom_sf(color = "white", size = 0.05) + 
  coord_sf(crs = 2992) + # Updated CRS specific to Northern Oregon / Portland
  scale_fill_distiller(palette = "YlGnBu", direction = 1) +
  theme_void() +
  labs(
    fill = "Local Segregation\nIndex (ls)",
    title = "Local Segregation Index by Census Tract",
    subtitle = "Multnomah County, OR (2016 ACS)",
    caption = "Sources: US Census Bureau, segregation R package"
  ) +
  theme(
    plot.title = element_text(face = "bold", hjust = 0.5),
    plot.subtitle = element_text(hjust = 0.5)
  )
